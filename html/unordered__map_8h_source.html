169 template<class K, class V, class Hash>
170 void unordered_map<K, V, Hash>::insert(const kv_map_t& nkv)
171 {
172  // resize if needed
173  if (m_element_count + 1 > m_size)
174  resize(m_size + 1);
175 
176  uint32_t index = hashing_function::run(nkv.first) % m_size;
177  unsigned int probe_len = 0;
178  unsigned int n_probe_len = 0;
179 
180  signed int swapIndex = -1;
181  while (probe_len < m_size)
182  {
183 
184  // if the object does not exist, emplcae it here
185  if (m_data.at(index).displacement == -1)
186  {
187  auto* _new = &m_data.at_m(index);
188  _new->first = nkv.first;
189  _new->second = nkv.second;
190  _new->displacement = probe_len;
191 
192  m_element_count++;
193 
194  if (swapIndex != -1)
195  {
196  swap(index, swapIndex, n_probe_len);
197  }
198 
199  return;
200  }
201  // if the keys match, update the value
202  else if (m_data.at(index).first == nkv.first)
203  {
204  m_data.at_m(index).second = nkv.second;
205  return;
206  }
207 
208  // linear probing
209  else
210  {
211  if (swapIndex == -1 && m_data.at(index).displacement < (int)probe_len)
212  {
213  swapIndex = index;
214  n_probe_len = probe_len;
215  }
216  index = (index + 1) % m_size;
217  }
218 
219  ++probe_len;
220 
221  }

267 {
268  int idx = index_of(k);
269  // no keys to delete
270  if (idx == -1)
271  return 0;
272 
274  kv_map_t* target = &m_data.at_m((size_t)idx);
275  int curpsl = target->displacement;
276  int iters = 0;
277 
278  // shift every item with a displacement higher than ours 1 index lower
279  while (m_data.at_m(++idx).displacement > curpsl && (iters < m_element_count))
280  {
281  auto* old = &m_data.at_m(idx);
282  old->displacement--;
283  memmove(target, old, sizeof(kv_map_t));
284 
285  target = &m_data.at_m(idx);
286 
287  idx = (idx + 1) % m_size;
288  ++iters;
289  }
290  target->displacement = -1;
291  --m_element_count;
292  return 1;
