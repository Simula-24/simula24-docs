161 void unordered_map<K, V, Hash>::insert(const kv_map_t& nkv)
162 {
163  // resize if needed
164  if (m_element_count + 1 > m_size)
165  resize(m_size + 1);
166 
167  uint32_t index = hashing_function::run(nkv.first) % m_size;
168  unsigned int probe_len = 0;
169  unsigned int n_probe_len = 0;
170 
171  signed int swapIndex = -1;
172  while (probe_len < m_size)
173  {
174 
175  // if the object does not exist, emplcae it here
176  if (m_data.at_p(index)->displacement == -1)
177  {
178  auto* _new = m_data.at_pc(index);
179  _new->first = nkv.first;
180  _new->second = nkv.second;
181  _new->displacement = probe_len;
182 
183  m_element_count++;
184 
185  if (swapIndex != -1)
186  {
187  swap(index, swapIndex, n_probe_len);
188  }
189 
190  return;
191  }
192  // if the keys match, update the value
193  else if (m_data.at(index).first == nkv.first)
194  {
195  m_data.at_c(index).second = nkv.second;
196  return;
197  }
198 
199  // linear probing
200  else
201  {
202  if (swapIndex == -1 && m_data.at(index).displacement < (int)probe_len)
203  {
204  swapIndex = index;
205  n_probe_len = probe_len;
206  }
207  index = (index + 1) % m_size;
208  }
